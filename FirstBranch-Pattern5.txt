//BUSINESS class (class for sub methods like multiple classes)
//Inheritance prgm using THIS()--> Statments and SUPER()--> Statement

class A
{
int i;
public A(int i)
    {
     this.i=i;
         }
}
class B extends A
{
     int j;
public B(int i,int j)
  {
   super(i);
   this.j=j;
   }
public void display()
   {
   System.out.println(this.i);
   System.out.println(this.j);
     }
}
class DriverClass1
{
public static void main(String[] args)
    {
       B b1 = new B(70,35);
       b1.display();
     }
}
